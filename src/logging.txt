[loggers]
keys=root

[handlers]
keys=fileHandler, consoleHandler

[formatters]
keys=consoleFormatter, defaultFormatter

[loggier_root]
level=INFO
handlers=consoleHandler,fileHandler
propagate=0

[handler_fileHandler]
class=handlers.TimeRotatingFileHandler
level=INFO
formatter=defaultFormatter
args=("./optuna.log", "midnight", 1, 30, "utf-8")

[handler_consoleHandler]
class=StreamHandler
level=INFO
formatter=consoleFormatter
args=(sys.stdout,)

[formatter_defaultFormatter]
format=%(asctime)s.%(msecs)03d | %(levelname)-8s | %(filename)s-%(funcName)s-%(lineno)04d | %(message)s
datefmt=%Y-%m-%d %H:%M:%S

[formatter_consoleFormatter]
format=%(asctime)s.%(msecs)03d | %(levelname)-8s | %(message)s
datefmt=%Y-%m-%d %H:%M:%S
